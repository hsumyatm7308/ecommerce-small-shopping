- Items 

CREATE TABLE IF NOT EXISTS items (
id INT AUTO_INCREMENT NOT NULL PRIMARY KEY,
image VARCHAR (255),
name VARCHAR(100),
price DECIMAL,
description VARCHAR(255),
brand_id INT,
category_id INT,
status_id INT,
user_id INT
);
